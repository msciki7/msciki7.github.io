`문제`

`코드`
```cpp

#include <stdio.h>
#include <math.h>
#define SEC_PER_MINUTE 60

	int a;
	scanf("%d", &a);

	printf("%d%d%d%d%d%d\n", a, a, a, a, a, a);
	printf("%d    %d\n", a, a);
	printf("%d    %d\n", a, a);
	printf("%d%d%d%d%d%d\n", a, a, a, a, a, a);

	return 0;
}
```
